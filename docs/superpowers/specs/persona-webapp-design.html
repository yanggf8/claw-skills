PRAGMA foreign_keys = ON;  -- 每次連線都需執行

CREATE TABLE user (
  id             INTEGER PRIMARY KEY,
  email          TEXT UNIQUE NOT NULL,          -- 存入前 toLowerCase()
  oauth_provider TEXT NOT NULL,                 -- 'google' | 'github'
  oauth_sub      TEXT NOT NULL,
  role           TEXT NOT NULL DEFAULT 'author',-- 'admin' | 'author'
  active         INTEGER NOT NULL DEFAULT 1,    -- 0 = deactivated
  created_at     TEXT NOT NULL,
  UNIQUE (oauth_provider, oauth_sub)
);

CREATE TABLE invite (
  id          INTEGER PRIMARY KEY,
  email       TEXT UNIQUE NOT NULL,             -- 存入前 toLowerCase()
  role        TEXT NOT NULL DEFAULT 'author',
  invited_by  INTEGER NOT NULL REFERENCES user(id),
  used_at     TEXT,                             -- NULL = 未使用
  created_at  TEXT NOT NULL
);

CREATE TABLE persona (
  slug               TEXT PRIMARY KEY,
  user_id            INTEGER NOT NULL REFERENCES user(id),
  name               TEXT NOT NULL,
  role               TEXT,
  expression         TEXT,
  mental_models      TEXT,
  heuristics         TEXT,
  antipatterns       TEXT,
  limits             TEXT,
  preview_text       TEXT,                      -- NULL = 尚未生成
  preview_updated_at TEXT,
  persona_updated_at TEXT                       -- 每次 UPDATE persona 時更新
);

CREATE TABLE persona_history (
  id           INTEGER PRIMARY KEY,
  persona_slug TEXT NOT NULL REFERENCES persona(slug),
  devto_id     TEXT,
  title        TEXT,
  published_at TEXT,
  status       TEXT NOT NULL DEFAULT 'published' -- 'published' | 'draft' | 'scheduled'
);

CREATE TABLE editorial_plan (
  id           INTEGER PRIMARY KEY,
  persona_slug TEXT NOT NULL REFERENCES persona(slug),
  angle        TEXT,
  lens         TEXT,
  direction    TEXT,
  key_question TEXT,
  status       TEXT NOT NULL DEFAULT 'draft',   -- 'draft' | 'active' | 'done'
  created_at   TEXT NOT NULL,
  updated_at   TEXT NOT NULL
);

CREATE TABLE audit_log (
  id         INTEGER PRIMARY KEY,
  user_id    INTEGER,                           -- NULL for service token calls
  action     TEXT NOT NULL,                     -- 'create' | 'update' | 'delete'
  entity     TEXT NOT NULL,
  entity_id  TEXT NOT NULL,
  caller     TEXT,                              -- 'user' | 'service/skill-name' (unverified for service)
  diff_json  TEXT,
  created_at TEXT NOT NULL
);

-- Indexes
CREATE INDEX idx_persona_user_id       ON persona(user_id);

Auth Middleware

// 兩條 auth path，互斥
app.use('/api/*', async (c, next) => {
  const token = c.req.header('Authorization')?.replace('Bearer ', '')
    ?? getCookie(c, 'session')

  // Path 1: Service Token（限定 2 個 write-back 端點）
  if (token && c.env.SERVICE_TOKEN) {
    const encoder = new TextEncoder()
    const a = encoder.encode(token)
    const b = encoder.encode(c.env.SERVICE_TOKEN)
    if (a.length === b.length &&
        crypto.subtle.timingSafeEqual(a, b)) {         // timing-safe
      const path = new URL(c.req.url).pathname
      const allowed = [/^\/api\/personas\/[^/]+\/preview$/, /^\/api\/personas\/[^/]+\/history$/]
      if (!allowed.some(r => r.test(path))) return c.json({ error: 'forbidden' }, 403)
      c.set('user', { role: 'service', sub: null,
                      caller: c.req.header('X-Service-Caller') ?? 'unknown' })
      await next(); return
    }
  }

  // Path 2: JWT（human users）
  const payload = await verifyJWT(token, c.env.JWT_SECRET)
  if (!payload) return c.json({ error: 'unauthorized' }, 401)
  if (!payload.active) return c.json({ error: 'account_deactivated' }, 403)
  c.set('user', payload)
  await next()
})

Section 5：寫作風格預覽機制 ✅ 確認

核心原則

  • CF Workers 不呼叫任何 LLM
  • Preview 由 skill（claw / cc）生成，model 與正式發佈一致
  • Webapp 只讀取 skill 寫回的結果

Section 6：錯誤處理與測試策略 ✅ 確認

Skill 端錯誤策略

  • exit 0，失敗印 [WARN: ...]，不 raise
  • LLM 失敗 → preview_text 維持 NULL，下次 cron 重試
  • API write-back 失敗 → log HTTP status + response body，skip

Service Token 注意事項

  • Token 存於 CF Workers Secret（wrangler secret put SERVICE_TOKEN）及 claw .env

  • X-Service-Caller header 為 unverified（audit log 標記為 "claimed"）

DB 備份

Turso 原生支援 point-in-time restore，但仍需主動備份以防邏輯刪除（誤操作）。

# backup.py — 可排入 claw cron，每日執行
import subprocess, datetime, os

TURSO_URL = os.environ["TURSO_URL"]
TURSO_TOKEN = os.environ["TURSO_AUTH_TOKEN"]
BACKUP_DIR = os.path.expanduser("~/.persona-backups")

def backup():
    os.makedirs(BACKUP_DIR, exist_ok=True)
    ts = datetime.datetime.utcnow().strftime("%Y%m%d-%H%M%S")
    out = f"{BACKUP_DIR}/persona-{ts}.sql"
    # 用 turso db shell 匯出 SQL dump
    subprocess.run(
        ["turso", "db", "shell", TURSO_URL, ".dump"],
        stdout=open(out, "w"), check=True,
        env={**os.environ, "TURSO_AUTH_TOKEN": TURSO_TOKEN}
    )
    print(f"[OK] backup → {out}")
    # 保留最近 30 份，刪除舊的
    files = sorted(os.listdir(BACKUP_DIR))
    for f in files[:-30]:
        os.remove(f"{BACKUP_DIR}/{f}")
